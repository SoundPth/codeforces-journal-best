#include <bits/stdc++.h>

using namespace std;

const int p = 31;
const int p1 = 37;
const int mod = 1e9 + 7;

struct Resid {
    int val;
};

Resid operator+(const Resid &a, const Resid &b) {
    int c = a.val + b.val;
    if (c >= mod) {
        return {c - mod};
    }
    return {c};
}

Resid operator-(const Resid &a, const Resid &b) {
    int c = a.val - b.val;
    if (c < 0) {
        return {c + mod};
    }
    return {c};
}

Resid get_char(char c) {
    return {c - 'a' + 1};
}

Resid operator*(const Resid &a, const Resid& b) {
    int c = (a.val * 1ll * b.val) % mod;
    return {c};
}

bool operator==(const Resid &a, const Resid &b) {
    return a.val == b.val;
}

Resid get_hash(int l, int r, vector<Resid>& hash, vector<Resid> &q) {
    Resid ans = hash[r] - hash[l] * q[r - l];
    return ans;
}
